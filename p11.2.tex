	\begin{algorithm}
		\caption{Find Maximum Value}
		\begin{algorithmic}[1]
			\REQUIRE A list of numbers $L = [l_1, l_2, \ldots, l_n]$
			\ENSURE The maximum value in the list
			\STATE Let $max \leftarrow L[0]$
			\FOR{each number $x$ in $L$}
			\IF{$x > max$}
			\STATE $max \leftarrow x$
			\ENDIF
			\ENDFOR
			\RETURN $max$
		\end{algorithmic}
	\end{algorithm}
